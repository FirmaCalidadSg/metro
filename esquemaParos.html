Diagrama Entidad-Relación y SQL

-- Tabla de Plantas CREATE TABLE Plantas ( id_planta INT PRIMARY KEY, nombre VARCHAR(50), ubicacion VARCHAR(100) ); -- Tabla de Categorías de Paros CREATE TABLE CategoriaParos ( id_categoria INT PRIMARY KEY, nombre VARCHAR(100), descripcion TEXT ); -- Tabla de Subcategorías CREATE TABLE SubcategoriaParos ( id_subcategoria INT PRIMARY KEY, id_categoria INT, nombre VARCHAR(100), descripcion TEXT, FOREIGN KEY (id_categoria) REFERENCES CategoriaParos(id_categoria) ); -- Tabla de Tipos de Paros CREATE TABLE TiposParos ( id_tipo INT PRIMARY KEY, id_subcategoria INT, codigo VARCHAR(20), descripcion VARCHAR(200), FOREIGN KEY (id_subcategoria) REFERENCES SubcategoriaParos(id_subcategoria) ); -- Tabla de Registro de Paros CREATE TABLE RegistroParos ( id_registro INT PRIMARY KEY, id_planta INT, id_tipo INT, fecha_inicio DATETIME, fecha_fin DATETIME, duracion INT, observaciones TEXT, FOREIGN KEY (id_planta) REFERENCES Plantas(id_planta), FOREIGN KEY (id_tipo) REFERENCES TiposParos(id_tipo) ); -- Tabla de Impacto de Paros CREATE TABLE ImpactoParos ( id_impacto INT PRIMARY KEY, id_registro INT, perdida_produccion DECIMAL(10,2), costo_estimado DECIMAL(10,2), FOREIGN KEY (id_registro) REFERENCES RegistroParos(id_registro) );
